# start of garden lighting config
base
  syntax_vsn 2020-0810
  timeout PT3M
  timezone America/New_York

devices
  porch pwm "front leds porch"
  red_maple pwm "front leds red maple"
  evergreen pwm "front leds evergreen"

cmds
  random_fade_bright random
    min 256
    max 2048
    primes 35
    step_ms 55
    step 31
    priority 7

  random_fade_dim random
    min 256
    max 1024
    primes 35
    step_ms 55
    step 13
    priority 7

jobs
  porch
    device porch
    schedule
      morning
        sun_ref civil_twilight_begin
        cmd off
      evening
        sun_ref sunset
        cmd random_fade_bright
      night
        sun_ref civil_twilight_end
        cmd random_fade_dim

  red_maple
    device red_maple
    schedule
      morning
        sun_ref civil_twilight_begin
        cmd off
      evening
        sun_ref sunset
        cmd random_fade_bright
      night
        sun_ref civil_twilight_end
        cmd random_fade_dim

  evergreen
    device evergreen
    schedule
      morning
        sun_ref civil_twilight_begin
        cmd off
      evening
        sun_ref sunset
        cmd random_fade_bright
      night
        sun_ref civil_twilight_end
        cmd random_fade_dim
